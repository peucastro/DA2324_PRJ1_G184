580 /*
581  * Performs a breadth-first search (bfs) in a graph (this), starting
582  * from the vertex with the given source contents (source).
583  * Returns a vector with the contents of the vertices by bfs order.
584  */
585 template <class T>
586 std::vector<T> Graph<T>::bfs(const T &source) const
587 {
588  std::vector<T> res;
589  // Get the source vertex
590  auto s = findVertex(source);
591  if (s == nullptr)
592  {
593  return res;
594  }
595 
596  // Set that no vertex has been visited yet
597  for (auto v : vertexSet)
598  {
599  v->setVisited(false);
600  }
601 
602  // Perform the actual BFS using a queue
603  std::queue<Vertex<T> *> q;
604  q.push(s);
605  s->setVisited(true);
606  while (!q.empty())
607  {
608  auto v = q.front();

610  res.push_back(v->getInfo());
611  for (auto &e : v->getAdj())
612  {
613  auto w = e->getDest();
614  if (!w->isVisited())
615  {
616  q.push(w);
617  w->setVisited(true);
618  }
619  }
620  }

676 /****************** toposort ********************/
677 /*
678  * Performs a topological sorting of the vertices of a graph (this).
679  * Returns a vector with the contents of the vertices by topological order.
680  * If the graph has cycles, returns an empty vector.
681  * Follows the algorithm described in theoretical classes.
682  */
683 
684 template <class T>
685 std::vector<T> Graph<T>::topsort() const
686 {
687  std::vector<T> res;
688 
689  for (auto v : vertexSet)
690  {
691  v->setIndegree(0);
692  }
693  for (auto v : vertexSet)
694  {
695  for (auto e : v->getAdj())
696  {
697  unsigned int indegree = e->getDest()->getIndegree();
698  e->getDest()->setIndegree(indegree + 1);
699  }
700  }
701 
702  std::queue<Vertex<T> *> q;
703  for (auto v : vertexSet)
704  {
705  if (v->getIndegree() == 0)
706  {
707  q.push(v);
708  }
709  }
710 
711  while (!q.empty())
712  {
713  Vertex<T> *v = q.front();
714  q.pop();
715  res.push_back(v->getInfo());
716  for (auto e : v->getAdj())
717  {
718  auto w = e->getDest();
719  w->setIndegree(w->getIndegree() - 1);
720  if (w->getIndegree() == 0)
721  {
722  q.push(w);
723  }
724  }
725  }
726 
727  if (res.size() != vertexSet.size())
728  {
729  // std::cout << "Impossible topological ordering!" << std::endl;
730  res.clear();
731  return res;
732  }
733 
734  return res;
735 }
